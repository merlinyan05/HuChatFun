# HuChatFun 数据清洗完整工作流

> 基于实际项目结构设计，语料来源：`HuChenFeng-1.1/` 目录下约 400+ 场直播文字稿

---

## 工作流总览

```
阶段0: 语料探查         →  搞清楚 .md 文件的实际格式、说话人标记方式
阶段1: 批量粗切          →  正则去噪，砍掉 60% 废话，输出干净文本
阶段2: 结构化切分        →  按连麦对话切分，输出独立对话段落 + 独白观点段
阶段3: 质量评分 & 过滤    →  自动打分，只保留高价值内容（约 15-20%）
阶段4: LLM 辅助构造      →  用 Claude API 把独白段配问题、把对话段整理成问答对
阶段5: 风格增强          →  用 Claude API 改写，口头禅密度拉满
阶段6: 格式化输出        →  导出 Qwen ChatML 格式的 train.json / eval.json
```

每个阶段都有明确的输入、输出、验收标准。

## 阶段 0：语料探查（手动，1-2 小时）

**目的**：搞清楚原始 .md 文件的实际格式，校准后续所有正则。

**操作**：
```bash
# 随机抽 5 个不同时期的文件看格式
# 早期（2023年）
cat HuChenFeng-1.1/2023年05月/2023-05-01.md | head -100

# 中期（2024年）  
cat HuChenFeng-1.1/2024年06月/2024-06-20.md | head -100

# 近期（2025年）
cat HuChenFeng-1.1/2025年09月/2025-09-08.md | head -100

# 带 INC 后缀的（可能是不完整的）
cat HuChenFeng-1.1/2023年06月/2023-06-15-1-INC.md | head -100
```

**要确认的事项清单**：

| 检查项 | 你需要记录的 |
|--------|-------------|
| 说话人标记格式 | `户晨风：` 还是 `**户晨风**：` 还是其他？ |
| 网友标记格式 | `某网友：` 还是有具体名字？ |
| 有没有时间戳 | `[00:12:34]` 格式？还是没有？ |
| markdown 格式元素 | 有没有 `#` 标题、`---` 分割线、`>` 引用？ |
| INC 文件的区别 | 是不完整的片段？还是分段？ |
| 不同时期格式是否一致 | 2023 和 2025 的格式有没有变化？ |
| 文件编码 | UTF-8？有没有乱码？ |
| 每个文件大概多长 | 几千字？几万字？ |

**输出**：一个 `format_notes.md`，记录上述所有发现，作为后续写正则的依据。

## 阶段 1：批量粗切（自动化脚本，跑一次约 5-10 分钟）

**输入**：`HuChenFeng-1.1/**/*.md`（约 400+ 个文件）
**输出**：`data/step1_rough_cleaned/` 目录下对应的 .txt 文件

```python
# code/step1_rough_clean.py

import re
from pathlib import Path
import json

INPUT_DIR = Path("HuChenFeng-1.1")
OUTPUT_DIR = Path("data/step1_rough_cleaned")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

# ====== 根据阶段0的探查结果调整这些正则 ======

def strip_markdown(text: str) -> str:
    """去掉 markdown 格式元素"""
    text = re.sub(r'^#{1,6}\s+.*$', '', text, flags=re.MULTILINE)  # 标题
    text = re.sub(r'^---+$', '', text, flags=re.MULTILINE)          # 分割线
    text = re.sub(r'^\*\*.*?\*\*', '', text, flags=re.MULTILINE)    # 加粗行
    text = re.sub(r'^>\s+', '', text, flags=re.MULTILINE)           # 引用
    return text

# 废话正则（按类别分组，方便调试）
NOISE_PATTERNS = {
    "感谢礼物": [
        r'感谢[我]?.*?[总种].*?(?:感谢|十分感谢|再次感谢)[。！]?',
        r'哎呀.*?(?:总|x总).*?(?:破费|花钱|消费|太感谢)',
        r'(?:又有|有[个]?)(?:舰长|提督|见长)',
        r'感谢[我]?.*?(?:的|啊)(?:舰长|提督|见长|SC|介绍|热气球|告白气球|'
        r'飞天|探索|风之吟唱|爱神|梦奇|红衣|急速|灯火|爱的乐章|心愿|'
        r'飞灯|基督|急速超跑|探头人强|红衣小电视)',
    ],
    "等人过渡": [
        r'好[，,]?\s*不着急啊[，,]?\s*不着急',
        r'稍安勿躁啊[，,]?\s*稍安勿躁',
        r'咱们稍微等一下人啊[，,]?\s*稍微等一下人',
        r'好[，,]?\s*不急啊[，,]?\s*不急',
        r'马上开始连[线麦]啊[，,]?\s*马上开始连[线麦]',
        r'大家重新申请一下',
        r'好[，,]?\s*下一个啊[，,]?\s*下一个',
        r'我现在把连麦[钮按].*?打开',
    ],
    "读SC过渡": [
        r'好[，,]?\s*(?:再)?[读对]一下SC啊',
        r'我先[去就]?看[微位私]',
        r'这个叫?xxxx总说',
        r'xxxx总说',
    ],
    "技术问题": [
        r'(?:怎么)?[有又]?卡[了呢死]',
        r'先给[对他]?方?闭[个一]?麦',
        r'你太卡了.*?知道吧',
        r'纯纯安卓网',
    ],
    "连麦操作": [
        r'点[最]?[左右上下].*?角.*?(?:加号|按钮|直播|关注)',
        r'你[能会知]道怎么开直播吗',
        r'输入我的名字户晨风',
        r'我给你点关注了',
        r'点一下[，,]?\s*往左边滑.*?开直播',
    ],
    "重复感谢名单": [
        r'(?:xxxx总[、，,]?\s*){3,}',
        r'(?:xxxx种[、，,]?\s*){3,}',
    ],
    "排队催促": [
        r'要排队啊[，,]?\s*[要]?排队',
        r'你别急[啊]?[，,]?\s*要排队',
    ],
}

def rough_clean(text: str) -> str:
    """粗切去噪"""
    text = strip_markdown(text)
    for category, patterns in NOISE_PATTERNS.items():
        for pattern in patterns:
            text = re.sub(pattern, '', text)
    # 清理残留空行
    text = re.sub(r'\n{3,}', '\n\n', text)
    text = '\n'.join(line.strip() for line in text.split('\n') if line.strip())
    return text

def process_all():
    stats = {"total_files": 0, "total_raw_chars": 0, "total_clean_chars": 0, "skipped": []}
    
    for md_file in sorted(INPUT_DIR.rglob("*.md")):
        # 跳过 README.md 等非语料文件
        if md_file.name in ("README.md", "SUMMARY.md", "Preface.md", 
                            "Acknowledgements.md", "videos.md"):
            continue
        
        stats["total_files"] += 1
        raw = md_file.read_text(encoding='utf-8')
        stats["total_raw_chars"] += len(raw)
        
        # 太短的可能是空文件或目录说明
        if len(raw) < 500:
            stats["skipped"].append(str(md_file))
            continue
        
        cleaned = rough_clean(raw)
        stats["total_clean_chars"] += len(cleaned)
        
        # 保持目录结构
        rel = md_file.relative_to(INPUT_DIR)
        out_path = OUTPUT_DIR / rel.with_suffix('.txt')
        out_path.parent.mkdir(parents=True, exist_ok=True)
        out_path.write_text(cleaned, encoding='utf-8')
    
    stats["noise_pct"] = f"{(1 - stats['total_clean_chars']/stats['total_raw_chars'])*100:.1f}%"
    
    # 保存统计
    (OUTPUT_DIR / "_stats.json").write_text(
        json.dumps(stats, ensure_ascii=False, indent=2), encoding='utf-8')
    
    print(f"处理文件: {stats['total_files']}")
    print(f"原始总字符: {stats['total_raw_chars']:,}")
    print(f"清洗后字符: {stats['total_clean_chars']:,}")
    print(f"去噪比例: {stats['noise_pct']}")
    print(f"跳过文件: {len(stats['skipped'])}")

if __name__ == "__main__":
    process_all()
```

**验收标准**：
- 去噪比例在 30-60% 之间（太低说明正则没写够，太高说明误删了）
- 随机抽 5 个输出文件，肉眼确认没有大段感谢/等人废话残留
- 随机抽 5 个输出文件，肉眼确认户晨风的观点段落没有被误删

## 阶段 2：结构化切分（自动化脚本）

**输入**：`data/step1_rough_cleaned/**/*.txt`
**输出**：`data/step2_segmented/conversations.jsonl` + `data/step2_segmented/monologues.jsonl`

```python
# code/step2_segment.py

import re
import json
from pathlib import Path
from dataclasses import dataclass, field, asdict

INPUT_DIR = Path("data/step1_rough_cleaned")
OUTPUT_DIR = Path("data/step2_segmented")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

@dataclass
class Turn:
    speaker: str
    text: str

@dataclass 
class Segment:
    source_file: str           # 来源文件
    segment_type: str          # "conversation" 或 "monologue"
    turns: list = field(default_factory=list)
    hu_text_length: int = 0    # 户晨风发言总字数

def split_by_speaker(text: str) -> list[Turn]:
    """按说话人切分（根据阶段0确认的实际格式调整正则）"""
    # 常见格式：户晨风：xxx 或 某网友：xxx
    pattern = r'^(户晨风|某网友)：'
    
    turns = []
    current_speaker = None
    current_lines = []
    
    for line in text.split('\n'):
        match = re.match(pattern, line)
        if match:
            if current_speaker and current_lines:
                turns.append(Turn(
                    speaker=current_speaker,
                    text='\n'.join(current_lines).strip()
                ))
            current_speaker = match.group(1)
            current_lines = [line[match.end():].strip()]
        elif current_speaker:
            current_lines.append(line)
    
    if current_speaker and current_lines:
        turns.append(Turn(
            speaker=current_speaker,
            text='\n'.join(current_lines).strip()
        ))
    
    return turns

def segment_file(filepath: Path) -> list[Segment]:
    """将单个文件切分为对话段和独白段"""
    text = filepath.read_text(encoding='utf-8')
    turns = split_by_speaker(text)
    
    if not turns:
        return []
    
    segments = []
    current = Segment(source_file=str(filepath), segment_type="conversation")
    
    # 新对话的信号
    NEW_SIGNALS = ['请讲', '你好你好', '直接表达观点', '说话', '速度']
    END_SIGNALS = ['再见', '拜拜', '生活愉快', '好了没有']
    
    for turn in turns:
        # 检测连麦开始
        is_new = (turn.speaker == '户晨风' and 
                  any(s in turn.text[:30] for s in NEW_SIGNALS) and
                  len(turn.text) < 100)
        
        if is_new and current.turns:
            current.hu_text_length = sum(
                len(t.text) for t in current.turns if t.speaker == '户晨风')
            segments.append(current)
            current = Segment(source_file=str(filepath), segment_type="conversation")
        
        current.turns.append(turn)
        
        # 检测连麦结束
        is_end = (turn.speaker == '户晨风' and
                  any(s in turn.text[-30:] for s in END_SIGNALS))
        
        if is_end:
            current.hu_text_length = sum(
                len(t.text) for t in current.turns if t.speaker == '户晨风')
            segments.append(current)
            current = Segment(source_file=str(filepath), segment_type="conversation")
    
    if current.turns:
        current.hu_text_length = sum(
            len(t.text) for t in current.turns if t.speaker == '户晨风')
        segments.append(current)
    
    # 分类：有网友参与的 = conversation，纯户晨风独白 = monologue
    for seg in segments:
        has_guest = any(t.speaker == '某网友' for t in seg.turns)
        seg.segment_type = "conversation" if has_guest else "monologue"
    
    return segments

def process_all():
    conversations = []
    monologues = []
    
    for txt_file in sorted(INPUT_DIR.rglob("*.txt")):
        if txt_file.name.startswith('_'):
            continue
        segments = segment_file(txt_file)
        for seg in segments:
            record = {
                "source": seg.source_file,
                "type": seg.segment_type,
                "hu_chars": seg.hu_text_length,
                "turn_count": len(seg.turns),
                "turns": [{"speaker": t.speaker, "text": t.text} for t in seg.turns],
            }
            if seg.segment_type == "conversation":
                conversations.append(record)
            else:
                monologues.append(record)
    
    # 写出
    with open(OUTPUT_DIR / "conversations.jsonl", 'w', encoding='utf-8') as f:
        for c in conversations:
            f.write(json.dumps(c, ensure_ascii=False) + '\n')
    
    with open(OUTPUT_DIR / "monologues.jsonl", 'w', encoding='utf-8') as f:
        for m in monologues:
            f.write(json.dumps(m, ensure_ascii=False) + '\n')
    
    print(f"连麦对话段: {len(conversations)}")
    print(f"独白观点段: {len(monologues)}")
    print(f"对话段平均轮数: {sum(c['turn_count'] for c in conversations)/max(len(conversations),1):.1f}")
    print(f"独白段平均字数: {sum(m['hu_chars'] for m in monologues)/max(len(monologues),1):.0f}")

if __name__ == "__main__":
    process_all()
```

**验收标准**：
- 对话段每段至少有 2 个 turn（一问一答）
- 独白段每段户晨风发言 > 50 字
- 总段落数合理（预计 3000-8000 段对话，1000-3000 段独白）

## 阶段 3：质量评分与过滤（自动化脚本）

**输入**：`data/step2_segmented/*.jsonl`
**输出**：`data/step3_filtered/high_quality.jsonl`

```python
# code/step3_score_and_filter.py

import json
from pathlib import Path

INPUT_DIR = Path("data/step2_segmented")
OUTPUT_DIR = Path("data/step3_filtered")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

# 高价值信号词（出现越多，分数越高）
HIGH_VALUE_SIGNALS = [
    # 观点输出标志
    "我告诉你", "我跟你这么讲", "我就跟你这样讲",
    "你给我记住了", "你记住这句话", "我就把话给你撂在这",
    "我再给你重复一遍",
    
    # 核心梗/口头禅
    "纯纯", "购买力", "前程似锦", "低论", "智商税",
    "安卓人", "安卓逻辑", "安卓学历", "安卓水", "安卓网",
    "苹果人",
    
    # 判断性语句
    "你废了", "你就是", "一定要", "绝对不要", "不要去",
    "有一个算一个", "大于等于",
    
    # 强调
    "这是常识", "这是事实", "这是医学常识",
    "能明白吗", "知道吧", "是不是", "对不对",
    "咱有什么说什么",
]

# 低价值信号（出现越多，扣分越多）
LOW_VALUE_SIGNALS = [
    "卡了", "闭麦", "开直播", "怎么开", "点一下",
    "等一下", "稍等", "不着急", "排队",
    "看微", "看私信", "通过", "下播",
    "感谢", "舰长", "提督", "SC",
]

# 话题标签
TOPIC_KEYWORDS = {
    "数码测评": ["iPhone", "苹果手机", "安卓手机", "华为", "特斯拉", "MacBook",
                "AirPods", "耳机", "路由器", "洗碗机", "单反", "相机"],
    "人生规划": ["学历", "工作", "大专", "本科", "985", "211", "雅思",
                "留学", "收入", "工资", "创业", "电焊", "offer"],
    "健康生活": ["减肥", "体检", "血压", "血脂", "医院", "中医",
                "筷子", "不锈钢", "木头", "发霉", "黄曲霉"],
    "消费观": ["山姆", "开市客", "德国", "洗碗机", "厨具", "菜市场",
              "超市", "品质", "品牌"],
    "价值观": ["贩卖焦虑", "底线", "离婚", "借钱", "拒绝",
              "明辨是非", "初心"],
    "房产城市": ["买房", "自建房", "租房", "大城市", "小地方",
                "别墅", "学区房", "农村"],
    "整理收纳": ["扔掉", "扔", "收拾", "整理", "玩偶", "架子"],
}

def score_segment(record: dict) -> dict:
    """给一个段落打分"""
    hu_text = ' '.join(
        t['text'] for t in record['turns'] if t['speaker'] == '户晨风'
    )
    
    score = 0
    reasons = []
    
    # === 加分项 ===
    
    # 1. 高价值信号词
    signal_hits = [s for s in HIGH_VALUE_SIGNALS if s in hu_text]
    signal_score = min(len(signal_hits), 5)  # 最多加5分
    score += signal_score
    if signal_hits:
        reasons.append(f"信号词x{len(signal_hits)}: {signal_hits[:3]}")
    
    # 2. 最佳长度区间
    if record['type'] == 'conversation':
        if 200 < record['hu_chars'] < 1500:
            score += 2
            reasons.append("长度适中")
        elif 100 < record['hu_chars'] <= 200:
            score += 1
    else:  # monologue
        if 150 < record['hu_chars'] < 800:
            score += 2
            reasons.append("长度适中")
    
    # 3. 对话轮数适中（对话段）
    if record['type'] == 'conversation' and 6 <= record['turn_count'] <= 40:
        score += 1
        reasons.append("轮数合理")
    
    # === 扣分项 ===
    
    # 4. 低价值信号
    low_hits = sum(1 for s in LOW_VALUE_SIGNALS if s in hu_text)
    if low_hits > 5:
        score -= 2
        reasons.append(f"低价值词多({low_hits})")
    
    # 5. 太短没内容
    if record['hu_chars'] < 50:
        score -= 3
        reasons.append("太短")
    
    # 6. 对话太短（上来就挂）
    if record['type'] == 'conversation' and record['turn_count'] < 4:
        score -= 2
        reasons.append("对话太短")
    
    # === 话题标签 ===
    topics = []
    for topic, keywords in TOPIC_KEYWORDS.items():
        if any(kw in hu_text for kw in keywords):
            topics.append(topic)
    
    record['score'] = max(0, score)
    record['reasons'] = reasons
    record['topics'] = topics if topics else ["其他"]
    
    return record

def process_all():
    results = {"high": [], "medium": [], "low": []}
    
    for jsonl_file in INPUT_DIR.glob("*.jsonl"):
        with open(jsonl_file, encoding='utf-8') as f:
            for line in f:
                record = json.loads(line)
                scored = score_segment(record)
                
                if scored['score'] >= 4:
                    results['high'].append(scored)
                elif scored['score'] >= 2:
                    results['medium'].append(scored)
                else:
                    results['low'].append(scored)
    
    # 输出高质量数据
    with open(OUTPUT_DIR / "high_quality.jsonl", 'w', encoding='utf-8') as f:
        for r in results['high']:
            f.write(json.dumps(r, ensure_ascii=False) + '\n')
    
    with open(OUTPUT_DIR / "medium_quality.jsonl", 'w', encoding='utf-8') as f:
        for r in results['medium']:
            f.write(json.dumps(r, ensure_ascii=False) + '\n')
    
    # 统计
    total = sum(len(v) for v in results.values())
    print(f"总段落数: {total}")
    print(f"高质量 (≥4分): {len(results['high'])} ({len(results['high'])/total*100:.1f}%)")
    print(f"中质量 (2-3分): {len(results['medium'])} ({len(results['medium'])/total*100:.1f}%)")
    print(f"低质量 (<2分):  {len(results['low'])} ({len(results['low'])/total*100:.1f}%)")
    
    # 话题分布
    from collections import Counter
    topic_counts = Counter()
    for r in results['high']:
        for t in r['topics']:
            topic_counts[t] += 1
    print("\n高质量段落话题分布:")
    for topic, count in topic_counts.most_common():
        print(f"  {topic}: {count}")

if __name__ == "__main__":
    process_all()
```

**验收标准**：
- 高质量段落占总量的 15-30%
- 随机抽 20 条高质量段落，肉眼确认确实有观点输出
- 随机抽 10 条低质量段落，肉眼确认确实是废话/无价值内容
- 如果高质量比例太低（<10%），放宽阈值到 ≥3 分
- 如果太高（>40%），说明去噪不够彻底，回阶段1加正则

## 阶段 4：LLM 辅助构造训练对（需要 API，耗时最长）

**输入**：`data/step3_filtered/high_quality.jsonl`
**输出**：`data/step4_training_pairs/pairs.jsonl`

这是整个流程中最关键也最费钱的一步。分两条路线处理：

### 路线 A：连麦对话 → 整理为多轮问答

```python
# code/step4a_conversations.py
"""
连麦对话本身就是问答格式，主要工作是：
1. 合并连续同一说话人的发言
2. 去掉对话中残留的废话（催促、技术问题）
3. 确保 user/assistant 严格交替
"""

import json
from pathlib import Path

INPUT = Path("data/step3_filtered/high_quality.jsonl")
OUTPUT = Path("data/step4_training_pairs")
OUTPUT.mkdir(parents=True, exist_ok=True)

SYSTEM_PROMPT = """你是户晨风（户子），B站数码科技直播博主。你的核心特征：

【说话方式】
- 上来先问关键信息：学历、收入、年龄、城市、工作、专业
- 口头禅密集："安卓"=贬义万能词，"苹果"=褒义万能词
- 高频使用："纯纯""我告诉你""你给我记住了""能明白吗""知道吧""咱有什么说什么"
- 用"购买力"衡量一切，用"前程似锦"反讽
- 说话极其直接，判断很绝对，从不模棱两可

【核心立场】
- 苹果生态绝对优于安卓，特斯拉绝对优于国产新能源
- 学历和收入高度正相关，低学历不努力=纯废
- 山姆/开市客 > 一切超市，德国厨具 > 一切
- 厨房必须全不锈钢，木制品全扔掉
- 年轻人必须去大城市，小地方没有未来
- 洗碗机是生活必需品，不买的人纯纯安卓思维
- 不借钱给任何人，能拒绝别人是能力"""

# 对话中需要清理的残留噪声
CONV_NOISE = [
    r'先给.*?闭麦',
    r'你太卡了',
    r'怎么[又有]?卡了',
    r'网[有点]?卡',
    r'先不着急',
]

def clean_conversation(record: dict) -> dict | None:
    """清理对话并格式化为训练数据"""
    turns = record['turns']
    
    # 合并连续同说话人发言 + 清理噪声
    merged = []
    for t in turns:
        text = t['text']
        # 去残留噪声
        import re
        for p in CONV_NOISE:
            text = re.sub(p, '', text)
        text = text.strip()
        if not text:
            continue
        
        role = "user" if t['speaker'] == '某网友' else "assistant"
        
        if merged and merged[-1]['role'] == role:
            merged[-1]['content'] += '\n' + text
        else:
            merged.append({"role": role, "content": text})
    
    # 确保以 user 开头
    while merged and merged[0]['role'] != 'user':
        merged.pop(0)
    
    # 确保交替（去重复角色）
    cleaned = []
    for t in merged:
        if not cleaned or t['role'] != cleaned[-1]['role']:
            cleaned.append(t)
    
    # 至少一轮问答
    if len(cleaned) < 2:
        return None
    
    return {
        "conversations": [
            {"role": "system", "content": SYSTEM_PROMPT},
            *cleaned
        ],
        "metadata": {
            "source": record['source'],
            "topics": record['topics'],
            "quality_score": record['score'],
            "type": "conversation",
        }
    }
```

### 路线 B：独白观点段 → 用 LLM 配问题

```python
# code/step4b_monologues.py
"""
独白段是户晨风自己讲的长段落（读SC回复、话题展开），
需要用 LLM 生成一个合理的"观众提问"来配对。

预算估算（Claude Sonnet）：
- 约 1000-2000 条独白
- 每条约 500 input tokens + 100 output tokens
- 总计约 100 万 tokens ≈ $3-5
"""

import anthropic
import json
import time
from pathlib import Path

client = anthropic.Anthropic()  # 从环境变量读 ANTHROPIC_API_KEY

INPUT = Path("data/step3_filtered/high_quality.jsonl")
OUTPUT = Path("data/step4_training_pairs/monologue_pairs.jsonl")

GEN_QUESTION_PROMPT = """你是一个数据标注助手。给你一段直播博主"户晨风"的发言，
请生成一个自然的观众提问，像真实观众在直播间连麦时会问的。

要求：
1. 简短口语化，一两句话
2. 要能自然引出博主的这段回答
3. 不要太具体（别把答案暗示在问题里）

博主发言：
{text}

只输出观众的提问，不要任何其他内容。"""

def generate_question(hu_text: str) -> str:
    """调 Claude 生成配对问题"""
    resp = client.messages.create(
        model="claude-sonnet-4-20250514",
        max_tokens=200,
        messages=[{
            "role": "user",
            "content": GEN_QUESTION_PROMPT.format(text=hu_text[:1500])
        }]
    )
    return resp.content[0].text.strip()

def process_monologues():
    records = []
    with open(INPUT, encoding='utf-8') as f:
        for line in f:
            r = json.loads(line)
            if r['type'] == 'monologue' and r['hu_chars'] > 100:
                records.append(r)
    
    print(f"待处理独白段: {len(records)}")
    
    with open(OUTPUT, 'w', encoding='utf-8') as out:
        for i, r in enumerate(records):
            hu_text = '\n'.join(
                t['text'] for t in r['turns'] if t['speaker'] == '户晨风'
            )
            
            try:
                question = generate_question(hu_text)
                pair = {
                    "conversations": [
                        {"role": "system", "content": SYSTEM_PROMPT},  # 同上
                        {"role": "user", "content": question},
                        {"role": "assistant", "content": hu_text},
                    ],
                    "metadata": {
                        "source": r['source'],
                        "topics": r['topics'],
                        "type": "monologue_with_generated_question",
                    }
                }
                out.write(json.dumps(pair, ensure_ascii=False) + '\n')
                
                if (i + 1) % 50 == 0:
                    print(f"进度: {i+1}/{len(records)}")
                
                time.sleep(0.5)  # 避免限速
                
            except Exception as e:
                print(f"跳过 {r['source']}: {e}")

if __name__ == "__main__":
    process_monologues()
```

## 阶段 6：格式化输出 + 划分数据集

**输入**：`data/step4_training_pairs/*.jsonl` + `data/step5_enhanced/*.jsonl`（如果做了）
**输出**：`data/final/train.json` + `data/final/eval.json`

```python
# code/step6_finalize.py

import json
import random
from pathlib import Path

INPUT_DIR = Path("data/step4_training_pairs")
ENHANCED_DIR = Path("data/step5_enhanced")  # 可能不存在
OUTPUT_DIR = Path("data/final")
OUTPUT_DIR.mkdir(parents=True, exist_ok=True)

def collect_all_pairs() -> list[dict]:
    """汇总所有训练对"""
    pairs = []
    
    for jsonl in INPUT_DIR.glob("*.jsonl"):
        with open(jsonl, encoding='utf-8') as f:
            for line in f:
                pairs.append(json.loads(line))
    
    if ENHANCED_DIR.exists():
        for jsonl in ENHANCED_DIR.glob("*.jsonl"):
            with open(jsonl, encoding='utf-8') as f:
                for line in f:
                    pairs.append(json.loads(line))
    
    return pairs

def validate_pair(pair: dict) -> bool:
    """验证单条训练数据的格式"""
    convs = pair.get('conversations', [])
    if len(convs) < 3:  # 至少 system + user + assistant
        return False
    if convs[0]['role'] != 'system':
        return False
    if convs[1]['role'] != 'user':
        return False
    # 检查 user/assistant 交替
    for i in range(2, len(convs)):
        expected = 'assistant' if i % 2 == 0 else 'user'
        if convs[i]['role'] != expected:
            return False
    # 最后一条必须是 assistant
    if convs[-1]['role'] != 'assistant':
        return False
    # assistant 回答不能太短
    if len(convs[-1]['content']) < 20:
        return False
    return True

def finalize():
    all_pairs = collect_all_pairs()
    print(f"汇总训练对: {len(all_pairs)}")
    
    # 验证格式
    valid = [p for p in all_pairs if validate_pair(p)]
    print(f"格式验证通过: {len(valid)} / {len(all_pairs)}")
    
    # 去重（按 assistant 最后一条回答的前 100 字去重）
    seen = set()
    unique = []
    for p in valid:
        key = p['conversations'][-1]['content'][:100]
        if key not in seen:
            seen.add(key)
            unique.append(p)
    print(f"去重后: {len(unique)}")
    
    # 随机打乱
    random.seed(42)
    random.shuffle(unique)
    
    # 90/10 划分
    split_idx = int(len(unique) * 0.9)
    train = unique[:split_idx]
    eval_set = unique[split_idx:]
    
    # 只保留 conversations 字段（去掉 metadata）
    def strip_meta(pairs):
        return [{"conversations": p["conversations"]} for p in pairs]
    
    with open(OUTPUT_DIR / "train.json", 'w', encoding='utf-8') as f:
        json.dump(strip_meta(train), f, ensure_ascii=False, indent=2)
    
    with open(OUTPUT_DIR / "eval.json", 'w', encoding='utf-8') as f:
        json.dump(strip_meta(eval_set), f, ensure_ascii=False, indent=2)
    
    print(f"\n最终输出:")
    print(f"  train.json: {len(train)} 条")
    print(f"  eval.json:  {len(eval_set)} 条")
    
    # 统计 token 数（粗略估算，中文约 1.5 字/token）
    total_chars = sum(
        sum(len(c['content']) for c in p['conversations'])
        for p in unique
    )
    est_tokens = int(total_chars / 1.5)
    print(f"  预估总 tokens: {est_tokens:,}")

if __name__ == "__main__":
    finalize()
```

## 关键提醒

1. **阶段 0 不能跳过**。你的 .md 文件格式可能和我假设的不一样，正则全靠这一步定。

2. **每个阶段之间都要手动抽查**。自动化脚本跑出来的东西，一定要肉眼看 20-50 条确认质量。

3. **阶段 5 可以先跳过**。先用阶段 4 的输出训一版模型试试效果，如果口头禅密度不够，再回来做增强。

4. **INC 文件要注意**。你目录里有很多 `-INC.md` 文件，可能是不完整的录制。阶段 0 时检查一下是否需要特殊处理或直接跳过。

5. **时间跨度带来的风格变化**。2023 年的户晨风和 2025 年的说话风格可能不一样（比如"安卓"梗可能是后来才多起来的）。如果你想要最新风格，可以给近期数据（2025）加权。
